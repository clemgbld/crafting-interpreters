class Doughnut {
  cook() {
    print "Fry until golden brown.";
    inner();
    print "Place in a nice box.";
  }
}

    cook() {
        print "Cookie monster.";
        inner();
    }
}

class BostonCream < Cookie {
  cook() {
    print "Pipe full of custard and coat with chocolate.";
  }
}
